Como configurar Database Modele no WildFly

1) Procure a tag <subsystem xmlns="urn:jboss:domain:security:1.2">

2) Dentro de <security-domains>, cole o seguinte trecho:

	<security-domain name="TremnCRMSecurityDomain" cache-type="default">
	    <authentication>
	        <login-module code="Database" flag="required">
	            <module-option name="dsJndiName" value="java:/PostgresDS"/>
	            <module-option name="principalsQuery" value="select u.password from Contact c, UserTremn u where u.contact_id=c.id and c.email=?"/>
	            <module-option name="rolesQuery" value="select u.profile, 'Roles' from UserTremn u, Contact c where u.contact_id=c.id and c.email=?"/>
	        </login-module>
	    </authentication>
	</security-domain>

3) Em web.xml, no final do arquivo, cole o seguinte trecho:

	<security-constraint>
		<web-resource-collection>
			<web-resource-name>Private Pages</web-resource-name>
			<url-pattern>/pages/*</url-pattern>
			<http-method>*</http-method>
		</web-resource-collection>
		<auth-constraint>
			<role-name>ADM</role-name>
			<role-name>OPE</role-name>
		</auth-constraint>
		<user-data-constraint>
			<transport-guarantee>NONE</transport-guarantee>
		</user-data-constraint>
	</security-constraint>

	<login-config>
		<auth-method>FORM</auth-method>
		<realm-name>TremnCRMSecurityDomain</realm-name>
		<form-login-config>
			<form-login-page>/login.xhtml</form-login-page>
			<form-error-page>/error502.xhtml</form-error-page>
		</form-login-config>
	</login-config>

	<security-role>
		<role-name>ADM</role-name>
	</security-role>
	<security-role>
		<role-name>OPE</role-name>
	</security-role>
